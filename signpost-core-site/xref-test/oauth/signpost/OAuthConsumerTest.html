89          HttpRequest request = mock(HttpRequest.class);
90          when(request.getRequestUrl()).thenReturn("http://example.com?a=1");
91          ByteArrayInputStream body = new ByteArrayInputStream("b=2".getBytes());
92          when(request.getMessagePayload()).thenReturn(body);
93          when(request.getContentType()).thenReturn(

95          when(request.getHeader("Authorization")).thenReturn(
96              "OAuth realm=www.example.com, oauth_signature=12345, oauth_version=1.1");
97  
98          OAuthMessageSigner signer = mock(HmacSha1MessageSigner.class);
99          consumer.setMessageSigner(signer);
100 
101         consumer.sign(request);
102 
103         // verify that all custom params are properly read and passed to the
104         // message signer
105         ArgumentMatcher<RequestParameters> hasAllParameters = new ArgumentMatcher<RequestParameters>() {
106             public boolean matches(Object argument) {
107                 RequestParameters params = (RequestParameters) argument;
108                 assertEquals("1", params.get("a").first());
109                 assertEquals("2", params.get("b").first());
110                 assertEquals("1.1", params.get("oauth_version").first());
111                 return true;
112             }
113         };
114 
115         verify(signer).sign(same(request), argThat(hasAllParameters));
116     }

164     private class IsCompleteListOfOAuthParameters extends
165             ArgumentMatcher<String> {
166 
167         @Override
168         public boolean matches(Object argument) {
169             String oauthHeader = (String) argument;
170             assertTrue(oauthHeader.startsWith("OAuth "));
171             assertAllOAuthParametersExist(OAuth.oauthHeaderToParamsMap(oauthHeader));
172             return true;
173         }
174     }

176     private void assertAllOAuthParametersExist(Map<String, String> params) {
