21      /***
22       * OAuth percent encoding demands characters from the URI reserved set to be
23       * percent encoded, and characters from the unreserved to NOT be percent
24       * encoded. All characters must be UTF-8 encoded first.
25       */
26      @Test
27      public void shouldHonorOAuthPercentEncodingRules() {
28  
29          // ALWAYS percent encode all characters from the reserved set as per RFC3986
30          assertEquals(reservedCharactersEncoded,
31                  OAuth.percentEncode(rfc3986ReservedCharacters));
32          // NEVER percent encode any characters from the unreserved set as per RFC3986
33          assertEquals(rfc3986UnreservedCharacters,
34                  OAuth.percentEncode(rfc3986UnreservedCharacters));
35          // percent encode spaces, do not use +
36          assertEquals("%20", OAuth.percentEncode(" "));
37          // percent encode %
38          assertEquals("%25", OAuth.percentEncode("%"));
39      }
40  
41      @Test
42      public void shouldCorrectlyPercentDecodeReservedCharacters() {
43          assertEquals(rfc3986ReservedCharacters,
44                  OAuth.percentDecode(reservedCharactersEncoded));
45          assertEquals("%", OAuth.percentDecode("%25"));
46          assertEquals(" ", OAuth.percentDecode("%20"));
47      }

49      @Test
50      public void shouldCorrectlyFormEncodeParameters() throws Exception {
51          HashMap<String, String> params = new HashMap<String, String>();
52          params.put("one", rfc3986ReservedCharacters);
53          params.put(rfc3986ReservedCharacters, rfc3986UnreservedCharacters);
54  
55          // both keys and values must be percent encoded

57                  + rfc3986UnreservedCharacters, OAuth.formEncode(params.entrySet()));
58      }
