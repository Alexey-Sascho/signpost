61      @Test
62      public void shouldNormalizeParameters() throws Exception {
63  
64          // should ignore signature, callback, and realm params
65          RequestParameters params = new RequestParameters();
66          params.put("a", "1");
67          params.put("realm", "www.example.com");
68          params.put("oauth_signature", "12345");
69          String result = new SignatureBaseString(httpGetMock, params).normalizeRequestParameters();
70          assertEquals("a=1", result);
71  
72          // example from OAuth spec
73          params = new RequestParameters();
74          params.put("a", "1");
75          params.put("c", "hi there");
76          params.put("f", "25");
77          params.put("f", "50");
78          params.put("f", "a");
79          params.put("z", "p");
80          params.put("z", "t");
81          String expected = "a=1&c=hi%20there&f=25&f=50&f=a&z=p&z=t";
82          result = new SignatureBaseString(httpGetMock, params).normalizeRequestParameters();
83          assertEquals(expected, result);
84  
85          // examples from the official test cases on
86          // http://oauth.pbwiki.com/TestCases
87          params = new RequestParameters();
88          params.put("a", "x!y");
89          params.put("a", "x y");
90          expected = "a=x%20y&a=x%21y";
91          result = new SignatureBaseString(httpGetMock, params).normalizeRequestParameters();
92          assertEquals(expected, result);
93  
94          params = new RequestParameters();
95          params.put("name", "");
96          assertEquals("name=", new SignatureBaseString(httpGetMock, params)
97              .normalizeRequestParameters());
98          params.putNull("name", null);
99          assertEquals("name=", new SignatureBaseString(httpGetMock, params)
100             .normalizeRequestParameters());
