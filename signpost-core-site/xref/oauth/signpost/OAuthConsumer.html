31  /***
32   * <p>
33   * Exposes a simple interface to sign HTTP requests using a given OAuth token
34   * and secret. Refer to {@link OAuthProvider} how to retrieve a valid token and
35   * token secret.
36   * </p>
37   * <p>
38   * HTTP messages are signed as follows:
39   * <p>
40   * 
41   * <pre>
42   * // exchange the arguments with the actual token/secret pair
43   * OAuthConsumer consumer = new DefaultOAuthConsumer(&quot;1234&quot;, &quot;5678&quot;);
44   * 
45   * URL url = new URL(&quot;http://example.com/protected.xml&quot;);
46   * HttpURLConnection request = (HttpURLConnection) url.openConnection();
47   * 
48   * consumer.sign(request);
49   * 
50   * request.connect();
51   * </pre>
52   * 
53   * </p>
54   * </p>
55   * 
56   * @author Matthias Kaeppler
57   */
58  public interface OAuthConsumer extends Serializable {
59  
60      /***
61       * Sets the message signer that should be used to generate the OAuth
62       * signature.
63       * 
64       * @param messageSigner
65       *        the signer
66       * @see HmacSha1MessageSigner
67       * @see PlainTextMessageSigner
68       */
69      public void setMessageSigner(OAuthMessageSigner messageSigner);
70  
71      /***
72       * Defines which strategy should be used to write a signature to an HTTP
73       * request.
74       * 
75       * @param signingStrategy
76       *        the strategy
77       * @see AuthorizationHeaderSigningStrategy
78       * @see QueryStringSigningStrategy
79       */
80      public void setSigningStrategy(SigningStrategy signingStrategy);
81  
82      /***
83       * <p>
84       * Causes the consumer to always include the oauth_token parameter to be
85       * sent, even if blank. If you're seeing 401s during calls to
86       * {@link OAuthProvider#retrieveRequestToken}, try setting this to true.
87       * </p>
88       * 
89       * @param enable
90       *        true or false
91       */
92      public void setSendEmptyTokens(boolean enable);
93  
94      /***
95       * Signs the given HTTP request by writing an OAuth signature (and other
96       * required OAuth parameters) to it. Where these parameters are written
97       * depends on the current {@link SigningStrategy}.
98       * 
99       * @param request
100      *        the request to sign
101      * @return the request object passed as an argument
102      * @throws OAuthMessageSignerException
103      * @throws OAuthExpectationFailedException
104      * @throws OAuthCommunicationException
105      */
106     public HttpRequest sign(HttpRequest request) throws OAuthMessageSignerException,
107             OAuthExpectationFailedException, OAuthCommunicationException;
108 
109     /***
110      * <p>
111      * Signs the given HTTP request by writing an OAuth signature (and other
112      * required OAuth parameters) to it. Where these parameters are written
113      * depends on the current {@link SigningStrategy}.
114      * </p>
115      * This method accepts HTTP library specific request objects; the consumer
116      * implementation must ensure that only those request types are passed which
117      * it supports.
118      * 
119      * @param request
120      *        the request to sign
121      * @return the request object passed as an argument
122      * @throws OAuthMessageSignerException
123      * @throws OAuthExpectationFailedException
124      * @throws OAuthCommunicationException
125      */
126 	public HttpRequest sign(Object request) throws OAuthMessageSignerException,
127             OAuthExpectationFailedException, OAuthCommunicationException;
128 
129     /***
130      * <p>
131      * "Signs" the given URL by appending all OAuth parameters to it which are
132      * required for message signing. The assumed HTTP method is GET.
133      * Essentially, this is equivalent to signing an HTTP GET request, but it
134      * can be useful if your application requires clickable links to protected
135      * resources, i.e. when your application does not have access to the actual
136      * request that is being sent.
137      * </p>
138      * 
139      * @param url
140      *        the input URL. May have query parameters.
141      * @return the input URL, with all necessary OAuth parameters attached as a
142      *         query string. Existing query parameters are preserved.
143      * @throws OAuthMessageSignerException
144      * @throws OAuthExpectationFailedException
145      * @throws OAuthCommunicationException
146      */
147     public String sign(String url) throws OAuthMessageSignerException,
148             OAuthExpectationFailedException, OAuthCommunicationException;
149 
150     /***
151      * Sets the OAuth token and token secret used for message signing.
152      * 
153      * @param token
154      *        the token
155      * @param tokenSecret
156      *        the token secret
157      */
158 	public void setTokenWithSecret(String token, String tokenSecret);
159 
160 	public String getToken();
161 
162 	public String getTokenSecret();
163 
164 	public String getConsumerKey();
165 
166 	public String getConsumerSecret();
167 
168     /***
169      * Returns all parameters collected from the HTTP request during message
170      * signing (this means the return value may be NULL before a call to
171      * {@link #sign}), plus all required OAuth parameters that were added
172      * because the request didn't contain them beforehand. In other words, this
173      * is the exact set of parameters that were used for creating the message
174      * signature.
175      * 
176      * @return the request parameters used for message signing
177      */
178     public RequestParameters getRequestParameters();
179 }
