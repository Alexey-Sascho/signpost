20  import oauth.signpost.basic.DefaultOAuthConsumer;

27  /***
28   * <p>
29   * Supplies an interface that can be used to retrieve request and access tokens
30   * from an OAuth 1.0(a) service provider. A provider object requires an
31   * {@link OAuthConsumer} to sign the token request message; after a token has
32   * been retrieved, the consumer is automatically updated with the token and the
33   * corresponding secret.
34   * </p>
35   * <p>
36   * To initiate the token exchange, create a new provider instance and configure
37   * it with the URLs the service provider exposes for requesting tokens and
38   * resource authorization, e.g.:
39   * </p>
40   * 
41   * <pre>
42   * OAuthProvider provider = new DefaultOAuthProvider(&quot;http://twitter.com/oauth/request_token&quot;,
43   *     &quot;http://twitter.com/oauth/access_token&quot;, &quot;http://twitter.com/oauth/authorize&quot;);
44   * </pre>
45   * <p>
46   * Depending on the HTTP library you use, you may need a different provider
47   * type, refer to the website documentation for how to do that.
48   * </p>
49   * <p>
50   * To receive a request token which the user must authorize, you invoke it using
51   * a consumer instance and a callback URL:
52   * </p>
53   * <p>
54   * 
55   * <pre>
56   * String url = provider.retrieveRequestToken(consumer, &quot;http://www.example.com/callback&quot;);
57   * </pre>
58   * 
59   * </p>
60   * <p>
61   * That url must be opened in a Web browser, where the user can grant access to
62   * the resources in question. If that succeeds, the service provider will
63   * redirect to the callback URL and append the blessed request token.
64   * </p>
65   * <p>
66   * That token must now be exchanged for an access token, as such:
67   * </p>
68   * <p>
69   * 
70   * <pre>
71   * provider.retrieveAccessToken(consumer, nullOrVerifierCode);
72   * </pre>
73   * 
74   * </p>
75   * <p>
76   * where nullOrVerifierCode is either null if your provided a callback URL in
77   * the previous step, or the pin code issued by the service provider to the user
78   * if the request was out-of-band (cf. {@link OAuth#OUT_OF_BAND}.
79   * </p>
80   * <p>
81   * The consumer used during token handshakes is now ready for signing.
82   * </p>
83   * 
84   * @see DefaultOAuthProvider
85   * @see DefaultOAuthConsumer
86   */
87  public interface OAuthProvider extends Serializable {
88  
89      /***
90       * Queries the service provider for a request token.
91       * <p>
92       * <b>Pre-conditions:</b> the given {@link OAuthConsumer} must have a valid
93       * consumer key and consumer secret already set.
94       * </p>
95       * <p>
96       * <b>Post-conditions:</b> the given {@link OAuthConsumer} will have an
97       * unauthorized request token and token secret set.
98       * </p>
99       * 
100      * @param consumer
101      *        the {@link OAuthConsumer} that should be used to sign the request
102      * @param callbackUrl
103      *        Pass an actual URL if your app can receive callbacks and you want
104      *        to get informed about the result of the authorization process.
105      *        Pass {@link OAuth.OUT_OF_BAND} if the service provider implements
106      *        OAuth 1.0a and your app cannot receive callbacks. Pass null if the
107      *        service provider implements OAuth 1.0 and your app cannot receive
108      *        callbacks. Please note that some services (among them Twitter)
109      *        will fail authorization if you pass a callback URL but register
110      *        your application as a desktop app (which would only be able to
111      *        handle OOB requests).
112      * @return The URL to which the user must be sent in order to authorize the
113      *         consumer. It includes the unauthorized request token (and in the
114      *         case of OAuth 1.0, the callback URL -- 1.0a clients send along
115      *         with the token request).
116      * @throws OAuthMessageSignerException
117      *         if signing the request failed
118      * @throws OAuthNotAuthorizedException
119      *         if the service provider rejected the consumer
120      * @throws OAuthExpectationFailedException
121      *         if required parameters were not correctly set by the consumer or
122      *         service provider
123      * @throws OAuthCommunicationException
124      *         if server communication failed
125      */
126     public String retrieveRequestToken(OAuthConsumer consumer, String callbackUrl)
127             throws OAuthMessageSignerException, OAuthNotAuthorizedException,
128             OAuthExpectationFailedException, OAuthCommunicationException;
129 
130     /***
131      * Queries the service provider for an access token.
132      * <p>
133      * <b>Pre-conditions:</b> the given {@link OAuthConsumer} must have a valid
134      * consumer key, consumer secret, authorized request token and token secret
135      * already set.
136      * </p>
137      * <p>
138      * <b>Post-conditions:</b> the given {@link OAuthConsumer} will have an
139      * access token and token secret set.
140      * </p>
141      * 
142      * @param consumer
143      *        the {@link OAuthConsumer} that should be used to sign the request
144      * @param oauthVerifier
145      *        <b>NOTE: Only applies to service providers implementing OAuth
146      *        1.0a. Set to null if the service provider is still using OAuth
147      *        1.0.</b> The verification code issued by the service provider
148      *        after the the user has granted the consumer authorization. If the
149      *        callback method provided in the previous step was
150      *        {@link OAuth.OUT_OF_BAND}, then you must ask the user for this
151      *        value. If your app has received a callback, the verfication code
152      *        was passed as part of that request instead.
153      * @throws OAuthMessageSignerException
154      *         if signing the request failed
155      * @throws OAuthNotAuthorizedException
156      *         if the service provider rejected the consumer
157      * @throws OAuthExpectationFailedException
158      *         if required parameters were not correctly set by the consumer or
159      *         service provider
160      * @throws OAuthCommunicationException
161      *         if server communication failed
162      */
163     public void retrieveAccessToken(OAuthConsumer consumer, String oauthVerifier)
164             throws OAuthMessageSignerException, OAuthNotAuthorizedException,
165             OAuthExpectationFailedException, OAuthCommunicationException;
166 
167     /***
168      * Any additional non-OAuth parameters returned in the response body of a
169      * token request can be obtained through this method. These parameters will
170      * be preserved until the next token request is issued. The return value is
171      * never null.
172      */
173     public Map<String, String> getResponseParameters();
174 
175     /***
176      * Subclasses must use this setter to preserve any non-OAuth query
177      * parameters contained in the server response. It's the caller's
178      * responsibility that any OAuth parameters be removed beforehand.
179      * 
180      * @param parameters
181      *        the map of query parameters served by the service provider in the
182      *        token response
183      */
184     public void setResponseParameters(Map<String, String> parameters);
185 
186     /***
187      * Use this method to set custom HTTP headers to be used for the requests
188      * which are sent to retrieve tokens.
189      * 
190      * @param header
191      *        The header name (e.g. 'WWW-Authenticate')
192      * @param value
193      *        The header value (e.g. 'realm=www.example.com')
194      */
195     public void setRequestHeader(String header, String value);
196 
197     /***
198      * @return all request headers set via {@link #setRequestHeader}
199      */
200     public Map<String, String> getRequestHeaders();
201 
202     /***
203      * @param isOAuth10aProvider
204      *        set to true if the service provider supports OAuth 1.0a. Note that
205      *        you need only call this method if you reconstruct a provider
206      *        object in between calls to retrieveRequestToken() and
207      *        retrieveAccessToken() (i.e. if the object state isn't preserved).
208      *        If instead those two methods are called on the same provider
209      *        instance, this flag will be deducted automatically based on the
210      *        server response during retrieveRequestToken(), so you can simply
211      *        ignore this method.
212      */
213     public void setOAuth10a(boolean isOAuth10aProvider);
214 
215     /***
216      * @return true if the service provider supports OAuth 1.0a. Note that the
217      *         value returned here is only meaningful after you have already
218      *         performed the token handshake, otherwise there is no way to
219      *         determine what version of the OAuth protocol the service provider
220      *         implements.
221      */
222     public boolean isOAuth10a();

224     public String getRequestTokenEndpointUrl();
