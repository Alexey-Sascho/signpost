15  package oauth.signpost.basic;
16  
17  import java.net.HttpURLConnection;
18  import java.net.URL;
19  import java.util.Map;
20  
21  import oauth.signpost.AbstractOAuthProvider;
22  import oauth.signpost.OAuth;
23  import oauth.signpost.OAuthConsumer;
24  import oauth.signpost.exception.OAuthCommunicationException;
25  import oauth.signpost.exception.OAuthExpectationFailedException;
26  import oauth.signpost.exception.OAuthMessageSignerException;
27  import oauth.signpost.exception.OAuthNotAuthorizedException;
28  import oauth.signpost.http.HttpRequest;
29  
30  /***
31   * This default implementation uses {@link java.net.HttpURLConnection} type GET
32   * requests to receive tokens from a service provider.
33   * 
34   * @author Matthias Kaeppler
35   */
36  public class DefaultOAuthProvider extends AbstractOAuthProvider {
37  
38      private static final long serialVersionUID = 1L;
39  
40      private HttpURLConnection connection;
41  
42      public DefaultOAuthProvider(String requestTokenEndpointUrl, String accessTokenEndpointUrl,
43              String authorizationWebsiteUrl) {
44          super(requestTokenEndpointUrl, accessTokenEndpointUrl, authorizationWebsiteUrl);
45      }
46  
47      protected void retrieveToken(OAuthConsumer consumer, String endpointUrl)
48              throws OAuthMessageSignerException, OAuthCommunicationException,
49              OAuthNotAuthorizedException, OAuthExpectationFailedException {
50  
51          Map<String, String> defaultHeaders = getRequestHeaders();
52  
53          if (consumer.getConsumerKey() == null || consumer.getConsumerSecret() == null) {
54              throw new OAuthExpectationFailedException("Consumer key or secret not set");
55          }
56  
57          try {
58              if (connection == null) {
59                  connection = (HttpURLConnection) new URL(endpointUrl).openConnection();
60                  connection.setRequestMethod("GET");
61              }
62              HttpRequest request = new HttpURLConnectionRequestAdapter(connection);
63              for (String header : defaultHeaders.keySet()) {
64                  request.setHeader(header, defaultHeaders.get(header));
65              }
66  
67              consumer.sign(request);
68  
69              connection.connect();
70  
71              int statusCode = connection.getResponseCode();
72  
73              if (statusCode == 401) {
74                  throw new OAuthNotAuthorizedException();
75              }
76  
77              Map<String, String> responseParams = OAuth.decodeForm(connection.getInputStream());
78  
79              String token = responseParams.get(OAuth.OAUTH_TOKEN);
80              String secret = responseParams.get(OAuth.OAUTH_TOKEN_SECRET);
81              responseParams.remove(OAuth.OAUTH_TOKEN);
82              responseParams.remove(OAuth.OAUTH_TOKEN_SECRET);
83  
84              setResponseParameters(responseParams);
85  
86              if (token == null || secret == null) {
87                  throw new OAuthExpectationFailedException(
88                      "Request token or token secret not set in server reply. "
89                              + "The service provider you use is probably buggy.");
90              }
91  
92              consumer.setTokenWithSecret(token, secret);
93  
94          } catch (OAuthNotAuthorizedException e) {
95              throw e;
96          } catch (OAuthExpectationFailedException e) {
97              throw e;
98          } catch (Exception e) {
99              throw new OAuthCommunicationException(e);
100         } finally {
101             if (connection != null) {
102                 connection.disconnect();
103                 connection = null;
104             }
105         }
106     }
107 
108     void setHttpUrlConnection(HttpURLConnection connection) {
109         this.connection = connection;
110     }
111 }
