77      public static String percentEncode(String s) {
78          if (s == null) {
79              return "";
80          }
81          return percentEncoder.escape(s);
82      }

84      public static String percentDecode(String s) {
85          try {
86              if (s == null) {
87                  return "";
88              }
89              return URLDecoder.decode(s, ENCODING);
90              // This implements http://oauth.pbwiki.com/FlexibleDecoding
91          } catch (java.io.UnsupportedEncodingException wow) {
92              throw new RuntimeException(wow.getMessage(), wow);
93          }
94      }

96      /***
97       * Construct a x-www-form-urlencoded document containing the given sequence
98       * of name/value pairs. Use OAuth percent encoding (not exactly the encoding
99       * mandated by x-www-form-urlencoded).
100      */
101     public static <T extends Map.Entry<String, String>> void formEncode(Collection<T> parameters,
102             OutputStream into) throws IOException {
103         if (parameters != null) {
104             boolean first = true;
105             for (Map.Entry<String, String> entry : parameters) {
106                 if (first) {
107                     first = false;
108                 } else {
109                     into.write('&');
110                 }
111                 into.write(percentEncode(safeToString(entry.getKey())).getBytes());
112                 into.write('=');
113                 into.write(percentEncode(safeToString(entry.getValue())).getBytes());
114             }
115         }
116     }

130     /*** Parse a form-urlencoded document. */
131     public static Map<String, String> decodeForm(String form) {
132         HashMap<String, String> params = new HashMap<String, String>();
133         if (isEmpty(form)) {
134             return params;
135         }
136         for (String nvp : form.split("//&")) {
137             int equals = nvp.indexOf('=');
138             String name;
139             String value;
140             if (equals < 0) {
141                 name = percentDecode(nvp);
142                 value = null;
143             } else {
144                 name = percentDecode(nvp.substring(0, equals));
145                 value = percentDecode(nvp.substring(equals + 1));
146             }
147 
148             params.put(name, value);
149         }
150         return params;
151     }

153     public static Map<String, String> decodeForm(InputStream content)
154             throws IOException {
155         BufferedReader reader = new BufferedReader(new InputStreamReader(
156                 content));
157         StringBuilder sb = new StringBuilder();
158         String line = reader.readLine();
159         while (line != null) {
160             sb.append(line);
161             line = reader.readLine();
162         }
163 
164         return decodeForm(sb.toString());
165     }

167     /***
168      * Construct a Map containing a copy of the given parameters. If several
169      * parameters have the same name, the Map will contain the first value,
170      * only.
171      */
172     public static <T extends Map.Entry<String, String>> Map<String, String> toMap(Collection<T> from) {
173         HashMap<String, String> map = new HashMap<String, String>();
174         if (from != null) {
175             for (Map.Entry<String, String> entry : from) {
176                 String key = entry.getKey();
177                 if (!map.containsKey(key)) {
178                     map.put(key, entry.getValue());
179                 }
180             }
181         }
182         return map;
183     }

193     /***
194      * Appends a list of key/value pairs to the given URL, e.g.:
195      * 
196      * <pre>
197      * String url = OAuth.addQueryParameters(&quot;http://example.com?a=1&quot;, b, 2, c, 3);
198      * </pre>
199      * 
200      * which yields:
201      * 
202      * <pre>
203      * http://example.com?a=1&b=2&c=3
204      * </pre>
205      * 
206      * All parameters will be encoded according to OAuth's percent encoding
207      * rules.
208      * 
209      * @param url
210      *        the URL
211      * @param kvPairs
212      *        the list of key/value pairs
213      * @return
214      */
215     public static String addQueryParameters(String url, String... kvPairs) {
216         String queryDelim = url.contains("?") ? "&" : "?";
217         StringBuilder sb = new StringBuilder(url + queryDelim);
218         for (int i = 0; i < kvPairs.length; i += 2) {
219             if (i > 0) {
220                 sb.append("&");
221             }
222             sb.append(OAuth.percentEncode(kvPairs[i]) + "="
223                     + OAuth.percentEncode(kvPairs[i + 1]));
224         }
225         return sb.toString();
226     }
227 
228     public static String addQueryParameters(String url, Map<String, String> params) {
229         String[] kvPairs = new String[params.size() * 2];
230         int idx = 0;
231         for (String key : params.keySet()) {
232             kvPairs[idx] = key;
233             kvPairs[idx + 1] = params.get(key);
234             idx += 2;
235         }
236         return addQueryParameters(url, kvPairs);
237     }

239     public static Map<String, String> oauthHeaderToParamsMap(String oauthHeader) {
240         if (oauthHeader == null || !oauthHeader.startsWith("OAuth ")) {
241             return new HashMap<String, String>();
242         }
243         oauthHeader = oauthHeader.substring("OAuth ".length());
244         String[] elements = oauthHeader.split(",");
245         HashMap<String, String> params = new HashMap<String, String>();
246         for (String keyValuePair : elements) {
247             String[] keyValue = keyValuePair.split("=");
248             params.put(keyValue[0].trim(), keyValue[1].trim());
249         }
250         return params;
251     }

253     /***
254      * Helper method to concatenate a parameter and its value to a pair that can
255      * be used in an HTTP header. This method percent encodes both parts before
256      * joining them.
257      * 
258      * @param name
259      *        the OAuth parameter name, e.g. oauth_token
260      * @param value
261      *        the OAuth parameter value, e.g. 'hello oauth'
262      * @return a name/value pair, e.g. oauth_token="hello%20oauth"
263      */
264     public static String toHeaderElement(String name, String value) {
265         return OAuth.percentEncode(name) + "=\"" + OAuth.percentEncode(value) + "\"";
