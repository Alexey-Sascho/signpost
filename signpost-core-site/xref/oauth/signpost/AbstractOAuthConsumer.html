34  /***
35   * ABC for consumer implementations. If you're developing a custom consumer you
36   * will probably inherit from this class to save you a lot of work.
37   * 
38   * @author Matthias Kaeppler
39   */
40  public abstract class AbstractOAuthConsumer implements OAuthConsumer {

42      private static final long serialVersionUID = 1L;

53      private RequestParameters requestParameters;

55      private boolean sendEmptyTokens;

57      public AbstractOAuthConsumer(String consumerKey, String consumerSecret) {
58          this.consumerKey = consumerKey;
59          this.consumerSecret = consumerSecret;
60          setMessageSigner(new HmacSha1MessageSigner());
61          setSigningStrategy(new AuthorizationHeaderSigningStrategy());
62      }
63  
64      public void setMessageSigner(OAuthMessageSigner messageSigner) {
65          this.messageSigner = messageSigner;
66          messageSigner.setConsumerSecret(consumerSecret);
67      }

69      public void setSigningStrategy(SigningStrategy signingStrategy) {
70          this.signingStrategy = signingStrategy;
71      }

73      public HttpRequest sign(HttpRequest request) throws OAuthMessageSignerException,
74              OAuthExpectationFailedException, OAuthCommunicationException {
75          if (consumerKey == null) {
76              throw new OAuthExpectationFailedException("consumer key not set");
77          }
78          if (consumerSecret == null) {
79              throw new OAuthExpectationFailedException("consumer secret not set");
80          }
81  
82          requestParameters = new RequestParameters();
83          try {
84              collectHeaderParameters(request, requestParameters);
85              collectQueryParameters(request, requestParameters);
86              collectBodyParameters(request, requestParameters);
87  
88              // add any OAuth params that haven't already been set
89              completeOAuthParameters(requestParameters);
90  
91              requestParameters.remove(OAuth.OAUTH_SIGNATURE);
92  
93          } catch (IOException e) {
94              throw new OAuthCommunicationException(e);
95          }
96  
97          String signature = messageSigner.sign(request, requestParameters);
98          OAuth.debugOut("signature", signature);
99  
100         signingStrategy.writeSignature(signature, request, requestParameters);
101 
102         return request;
103     }

105 	public HttpRequest sign(Object request) throws OAuthMessageSignerException,
106             OAuthExpectationFailedException, OAuthCommunicationException {
107 		return sign(wrap(request));
108 	}

110     public String sign(String url) throws OAuthMessageSignerException,
111             OAuthExpectationFailedException, OAuthCommunicationException {
112         HttpRequest request = new UrlStringRequestAdapter(url);
113 
114         // switch to URL signing
115         SigningStrategy oldStrategy = this.signingStrategy;
116         this.signingStrategy = new QueryStringSigningStrategy();
117 
118         sign(request);
119 
120         // revert to old strategy
121         this.signingStrategy = oldStrategy;
122 
123         return request.getRequestUrl();
124     }

157     /***
158      * <p>
159      * Helper method that adds any OAuth parameters to the given request
160      * parameters which are missing from the current request but required for
161      * signing. A good example is the oauth_nonce parameter, which is typically
162      * not provided by the client in advance.
163      * </p>
164      * <p>
165      * It's probably not a very good idea to override this method. If you want
166      * to generate different nonces or timestamps, override
167      * {@link #generateNonce()} or {@link #generateTimestamp()} instead.
168      * </p>
169      * 
170      * @param out
171      *        the request parameter which should be completed
172      */
173     protected void completeOAuthParameters(RequestParameters out) {
174         if (!out.containsKey(OAuth.OAUTH_CONSUMER_KEY)) {
175             out.put(OAuth.OAUTH_CONSUMER_KEY, consumerKey);
176         }
177         if (!out.containsKey(OAuth.OAUTH_SIGNATURE_METHOD)) {
178             out.put(OAuth.OAUTH_SIGNATURE_METHOD, messageSigner.getSignatureMethod());
179         }
180         if (!out.containsKey(OAuth.OAUTH_TIMESTAMP)) {
181             out.put(OAuth.OAUTH_TIMESTAMP, generateTimestamp());
182         }
183         if (!out.containsKey(OAuth.OAUTH_NONCE)) {
184             out.put(OAuth.OAUTH_NONCE, generateNonce());
185         }
186         if (!out.containsKey(OAuth.OAUTH_VERSION)) {
187             out.put(OAuth.OAUTH_VERSION, OAuth.VERSION_1_0);
188         }
189         if (!out.containsKey(OAuth.OAUTH_TOKEN)) {
190             if (token != null && !token.equals("") || sendEmptyTokens) {
191                 out.put(OAuth.OAUTH_TOKEN, token);
192             }
193         }
194     }

204     /***
205      * Collects OAuth Authorization header parameters as per OAuth Core 1.0 spec
206      * section 9.1.1
207      */
208     protected void collectHeaderParameters(HttpRequest request, RequestParameters out) {
209         Map<String, String> headerParams = OAuth.oauthHeaderToParamsMap(request
210             .getHeader(OAuth.HTTP_AUTHORIZATION_HEADER));
211         out.putMap(headerParams);
212     }
213 
214     /***
215      * Collects x-www-form-urlencoded body parameters as per OAuth Core 1.0 spec
216      * section 9.1.1
217      */
218     protected void collectBodyParameters(HttpRequest request, RequestParameters out)
219             throws IOException {
220 
221         // collect x-www-form-urlencoded body params
222         String contentType = request.getContentType();
223         if (contentType != null && contentType.startsWith(OAuth.FORM_ENCODED)) {
224             InputStream payload = request.getMessagePayload();
225             out.putMap(OAuth.decodeForm(payload));
226         }
227     }
228 
229     /***
230      * Collects HTTP GET query string parameters as per OAuth Core 1.0 spec
231      * section 9.1.1
232      */
233     protected void collectQueryParameters(HttpRequest request, RequestParameters out) {
234 
235         String url = request.getRequestUrl();
236         int q = url.indexOf('?');
237         if (q >= 0) {
238             // Combine the URL query string with the other parameters:
239             out.putMap(OAuth.decodeForm(url.substring(q + 1)));
240         }
241     }
242 
243     protected String generateTimestamp() {
244         return Long.toString(System.currentTimeMillis() / 1000L);
245     }
246 
247     protected String generateNonce() {
248         return Long.toString(new Random().nextLong());
249     }
