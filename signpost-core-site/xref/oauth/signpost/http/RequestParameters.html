13  /***
14   * A multi-map of HTTP request parameters. Each key references a
15   * {@link SortedSet} of parameters collected from the request during message
16   * signing. Parameter values are sorted as per {@linkplain http
17   * ://oauth.net/core/1.0a/#anchor13}. Every key/value pair will be
18   * percent-encoded upon insertion. This class therefore has very special
19   * semantics tailored to being useful for message signing. It's not a general
20   * purpose collection class to handle request parameters.
21   * 
22   * @author Matthias Kaeppler
23   */

31          return wrappedMap.put(key, value);
32      }

34      /***
35       * Convenience method to add a single value for the parameter specified by
36       * 'key'.
37       * 
38       * @param key
39       *        the parameter name
40       * @param value
41       *        the parameter value
42       * @return the value
43       */
44      public String put(String key, String value) {
45          SortedSet<String> values = wrappedMap.get(key);
46          if (values == null) {
47              values = new TreeSet<String>();
48              wrappedMap.put(OAuth.percentEncode(key), values);
49          }
50          value = OAuth.percentEncode(value);
51          values.add(value);
52  
53          return value;
54      }

90      /***
91       * Returns an application/x-www-form-encoded string of all values for the
92       * given parameter.
93       * 
94       * @param key
95       *        the parameter name
96       * @return the form encoded value string
97       */
98      public String getFormEncoded(Object key) {
99          StringBuilder sb = new StringBuilder();
100         key = OAuth.percentEncode((String) key);
101         Set<String> values = wrappedMap.get(key);
102         if (values == null) {
103             return key + "=";
104         }
105         Iterator<String> iter = values.iterator();
106         while (iter.hasNext()) {
107             sb.append(key + "=" + iter.next());
108             if (iter.hasNext()) {
109                 sb.append("&");
110             }
111         }
112         return sb.toString();
113     }
