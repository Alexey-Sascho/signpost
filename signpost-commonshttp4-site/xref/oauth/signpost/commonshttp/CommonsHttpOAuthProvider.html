16  
17  import java.io.IOException;
18  import java.util.Map;
19  
20  import oauth.signpost.AbstractOAuthProvider;
21  import oauth.signpost.OAuth;
22  import oauth.signpost.OAuthConsumer;
23  import oauth.signpost.exception.OAuthCommunicationException;
24  import oauth.signpost.exception.OAuthExpectationFailedException;
25  import oauth.signpost.exception.OAuthMessageSignerException;
26  import oauth.signpost.exception.OAuthNotAuthorizedException;
27  
28  import org.apache.http.HttpEntity;
29  import org.apache.http.HttpResponse;
30  import org.apache.http.client.HttpClient;
31  import org.apache.http.client.methods.HttpGet;
32  import org.apache.http.impl.client.DefaultHttpClient;
33  
34  /***
35   * This implementation uses the Apache Commons {@link HttpClient} 4.x HTTP
36   * implementation to fetch OAuth tokens from a service provider. Android users
37   * should use this provider implementation in favor of the default one, since
38   * the latter is known to cause problems with Android's Apache Harmony
39   * underpinnings.
40   * 
41   * @author Matthias Kaeppler
42   */
43  public class CommonsHttpOAuthProvider extends AbstractOAuthProvider {
44  
45      private static final long serialVersionUID = 1L;
46  
47      private HttpClient httpClient;
48  
49      public CommonsHttpOAuthProvider(String requestTokenEndpointUrl, String accessTokenEndpointUrl,
50              String authorizationWebsiteUrl) {
51          super(requestTokenEndpointUrl, accessTokenEndpointUrl, authorizationWebsiteUrl);
52          httpClient = new DefaultHttpClient();
53      }
54      
55      protected void retrieveToken(OAuthConsumer consumer, String endpointUrl)
56              throws OAuthMessageSignerException, OAuthCommunicationException,
57              OAuthNotAuthorizedException, OAuthExpectationFailedException {
58  
59          Map<String, String> defaultHeaders = getRequestHeaders();
60  
61          if (consumer.getConsumerKey() == null || consumer.getConsumerSecret() == null) {
62              throw new OAuthExpectationFailedException("Consumer key or secret not set");
63          }
64  
65          HttpGet request = new HttpGet(endpointUrl);
66          for (String header : defaultHeaders.keySet()) {
67              request.setHeader(header, defaultHeaders.get(header));
68          }
69          HttpResponse response = null;
70  
71          try {
72  
73              consumer.sign(new HttpRequestAdapter(request));
74  
75              response = httpClient.execute(request);
76  
77              int statusCode = response.getStatusLine().getStatusCode();
78  
79              if (statusCode == 401) {
80                  throw new OAuthNotAuthorizedException();
81              }
82  
83              Map<String, String> responseParams = OAuth
84                  .decodeForm(response.getEntity().getContent());
85  
86              String token = responseParams.get(OAuth.OAUTH_TOKEN);
87              String secret = responseParams.get(OAuth.OAUTH_TOKEN_SECRET);
88              responseParams.remove(OAuth.OAUTH_TOKEN);
89              responseParams.remove(OAuth.OAUTH_TOKEN_SECRET);
90  
91              setResponseParameters(responseParams);
92  
93              if (token == null || secret == null) {
94                  throw new OAuthExpectationFailedException(
95                      "Request token or token secret not set in server reply. "
96                              + "The service provider you use is probably buggy.");
97              }
98  
99              consumer.setTokenWithSecret(token, secret);
100 
101         } catch (OAuthNotAuthorizedException e) {
102             throw e;
103         } catch (OAuthExpectationFailedException e) {
104             throw e;
105         } catch (Exception e) {
106             throw new OAuthCommunicationException(e);
107         } finally {
108             if (response != null) {
109                 HttpEntity entity = response.getEntity();
110                 if (entity != null) {
111                     try {
112                         // free the connection
113                         entity.consumeContent();
114                     } catch (IOException e) {
115                         // this means HTTP keep-alive is not possible
116                         e.printStackTrace();
117                     }
118                 }
119             }
120         }
121     }
122 
123     void setHttpClient(HttpClient httpClient) {
124         this.httpClient = httpClient;
125     }
126 }
